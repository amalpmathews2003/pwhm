89/*
90 * @brief Handler to prepare custom evt msg processing:
91 * by fetching the real source interface and/or modify the received msg data to allow a generic processing.
92 * When needed, newInterfaceName and newMsgData are allocated and filled
93 * inside the handler, and they are freed on pwhm side
94 * This handler allows to redirect custom wpactrl messages received on a global wpaCtrl socket
95 * but targeted to a different interface
96 *
97 * @param userData private user data context registered with the wpactrl iface socket
98 * @param ifName name of wpactrl iface receiving the event
99 * @param msgData original text of the received msg
100 * @param len original length of received msg
101 * @param newIfName pointer to string allocated and filled with the interface name fetched in the msg
102 * @param newMsgData pointer to string allocated and filled with the prepared msg for processing
103 * @param newLen pointer to length variable set with the length of the prepared msg
104 *
105 * @return SWL_RC_DONE if the msg or the interface has been modified
106 * SWL_RC_OK if no custom changes has been done on target interface or received msg
107 * error code otherwise
108 * @return void
109 */
110typedef swl_rc_ne (* wld_wpaCtrl_preProcMsgCb_f)(void* userData, char* ifName, char* msgData, size_t len, char** newIfName, char** newMsgData, size_t* newLen);

277/*
278 * @brief parse a wpactrl msg and fetch the event name from a provided list
279 * then copy the event name (delimited with separator), and the argument list (starting with separator)
280 * The event name is potentially prefixed.
281 * The expected message format is "[PREFIX]<EVENT_NAME><separator><EVENT_ARGS...>"
282 *
283 * @param pData wp ctrl message
284 * @param prefix optional string prefixing the event name (typically the loglevel id , eg: <3>)
285 * @param separator optional string (or char) separating event name and next event arguments
286 * @param evtList optional array of event name strings to fetch in priority, before splitting msg with separator
287 * @param evtListLen length of optional array of known event name strings
288 * @param pEvtName pointer to output string allocated and filled with the event name (to be freed by the caller)
289 * @param pEvtArgs pointer to output string allocated and filled with the event arguments (to be freed by the caller)
290 *
291 * @return -1, when event name is unknown or not found
292 * index < evtListLen, when event name is found in the provided list
293 */
294int32_t wld_wpaCtrl_fetchEvent(const char* pData, const char* prefix, const char* sep, char* evtList[], size_t evtListLen, char** pEvtName, char** pEvtArgs);
