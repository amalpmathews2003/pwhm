76/*
77 * @brief return registered handlers table, defining wld implementation for nl80211
78 *
79 * @return pointer to function table
80 * or NULL if nl80211 implementation was not registered
81 */
82const T_CWLD_FUNC_TABLE* wld_nl80211_getVendorTable();
83
84#define CALL_NL80211_FTA(fun, arg, ...) \
85 { \
86 const T_CWLD_FUNC_TABLE* ftaTab = wld_nl80211_getVendorTable(); \
87 if((ftaTab != NULL) && (ftaTab->fun != NULL)) { \
88 ftaTab->fun(arg, ## __VA_ARGS__); \
89 } \
90 }
91
92#define CALL_NL80211_FTA_RET(ret, fun, arg, ...) \
93 { \
94 ret = SWL_RC_NOT_IMPLEMENTED; \
95 const T_CWLD_FUNC_TABLE* ftaTab = wld_nl80211_getVendorTable(); \
96 if((ftaTab != NULL) && (ftaTab->fun != NULL)) { \
97 ret = ftaTab->fun(arg, ## __VA_ARGS__); \
98 } \
99 }

101/*
102 * @brief convert time unit into mseconds: 100 TU (102.4ms).
103 * Ref: https://git.kernel.org/pub/scm/linux/kernel/git/torvalds/linux.git/tree/include/uapi/linux/nl80211.h#n648
104 */
105static inline uint64_t wld_nl80211_tu2ms(uint64_t tu) {
106 return ((tu * 1024) / 1000);
107}

117/*
118 * @brief return registered FSM manager, defining action sequences and state transitions
119 * of wld implementation for nl80211.
120 * The FSM manager defines the way the scheduled action bitmaps are applied,
121 * and also the dependencies between the scheduled actions for Radio, AccessPoint or Endpoint.
122 *
123 * Technically, every vendor module may (and shall):
124 * - define its own STATIC fsm manager, with defining its own action bitmaps
125 * to be scheduled when the function table handlers are called.
126 * The FSM manager relies on generic wld FSM implementation
127 * - share the same FSM manager as nl80211 default implementation
128 * when calling wld_fsm_init, while registering the new vendor to wld.
129 *
130 * @return pointer to FSM manager
131 * or NULL if nl80211 implementation was not registered
132 */
