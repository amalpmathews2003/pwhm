132/*
133 * @brief macro to initialize nl attributes list with array of attribute values
134 */
135#define NL_ATTRS_SET(pAttrList, attrValues) \
136 wld_nl80211_initNlAttrList(pAttrList); \
137 do { \
138 wld_nl80211_nlAttr_t attrArray[] = attrValues; \
139 for(uint32_t i = 0; i < SWL_ARRAY_SIZE(attrArray); i++) { \
140 swl_unLiList_add(pAttrList, &(attrArray[i])); \
141 }; \
142 } while(0);
143
144/*
145 * @brief macro to add nl attribute to attributes list
146 */
147#define NL_ATTRS_ADD(pAttrList, attrVal) \
148 do { \
149 wld_nl80211_nlAttr_t attr = attrVal; \
150 swl_unLiList_add(pAttrList, &attr); \
151 } while(0);
152
153/*
154 * @brief macro to define nl attributes list and fill it with array of attribute values
155 */
156#define NL_ATTRS(attrList, attrValues) \
157 wld_nl80211_nlAttrList_t attrList; \
158 wld_nl80211_initNlAttrList(&attrList); \
159 do { \
160 wld_nl80211_nlAttr_t attrArray[] = attrValues; \
161 for(uint32_t i = 0; i < SWL_ARRAY_SIZE(attrArray); i++) { \
162 swl_unLiList_add(&attrList, &(attrArray[i])); \
163 }; \
164 } while(0);
