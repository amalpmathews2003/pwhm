70/*
71 * @brief nl80211 event listener context

73struct wld_nl80211_listener {
74 amxc_llist_it_t it; //iterator for list
75 wld_nl80211_state_t* state; //manager ctx, receiving nl events
76 uint32_t wiphy; //watched wiphy: >=0 or WLD_NL80211_ID_ANY or WLD_NL80211_ID_UNDEF

78 void* pRef; //user private reference to provide when invoking handlers

81};

83/*
84 * @brief prototype of nl event parser: gets attributes from nl msg then forward event
85 * to the listener side
86 *
87 * @param pListener pointer to event listener
88 * @param nlh pointer to netlink msg header
89 * @param tb array of netlink message attributes
90 *
91 * @return SWL_RC_OK in case of parsing success
92 * SWL_RC_CONTINUE when msg is valid but ignored by the parser
93 * SWL_RC_DONE msg parsed and processed by listener event handler
94 * SWL_RC_ERROR in case of error
95 */
96typedef swl_rc_ne (* wld_nl80211_evtParser_f) (swl_unLiList_t* pListenerList, struct nlmsghdr* nlh, struct nlattr* tb[]);
97
98/*
99 * @brief get parser (callback) for nl80211 event
100 *
101 * @param eventId id of nl80211 event (NL80211_CMD_xxx)
102 *
103 * @return pointer to parser when event is supported, null otherwise
104 */
105wld_nl80211_evtParser_f wld_nl80211_getEventParser(uint32_t eventId);
106
107/*
108 * @brief check if listener has handler callback for received event
109 *
110 * @param pListener pointer to listener context
111 * @param eventId id of received nl80211 event (NL80211_CMD_xxx)
112 *
113 * @return true if listener has handler for received event
114 * false otherwise (or in case of error)
115 */
116bool wld_nl80211_hasEventHandler(wld_nl80211_listener_t* pListener, uint32_t eventId);
117
118/*
119 * @brief update event handlers of listener, when it defines
120 * monitored wiphy Id and/or iface netId.
121 *
122 * @param pListener pointer to listener context

124 * unwanted events shall remain null.
125 *
126 * @return SWL_RC_OK in case of success
127 * SWL_RC_ERROR otherwise
128 */
129swl_rc_ne wld_nl80211_updateEventHandlers(wld_nl80211_listener_t* pListener, const wld_nl80211_evtHandlers_cb* const handlers);
