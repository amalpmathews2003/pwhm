97#define CALL_MGR_EXT(pMgr, fName, ifName, ...) \
98 { \
99 void* userdata = NULL; \
100 wld_wpaCtrl_radioEvtHandlers_cb handlers = {0}; \
101 if(wld_wpaCtrlMngr_getEvtHandlers(pMgr, &userdata, &handlers)) { \
102 SWL_CALL(handlers.fName, userdata, ifName, __VA_ARGS__); \
103 } \
104 }
105
106#define CALL_MGR_I_EXT(pIntf, fName, ...) \
107 if(pIntf != NULL) { \
108 CALL_MGR_EXT(wld_wpaCtrlInterface_getMgr(pIntf), fName, wld_wpaCtrlInterface_getName(pIntf), __VA_ARGS__); \
109 }
110
111#define CALL_MGR_NA_EXT(pMgr, fName, ifName) \
112 { \
113 void* userdata = NULL; \
114 wld_wpaCtrl_radioEvtHandlers_cb handlers = {0}; \
115 if(wld_wpaCtrlMngr_getEvtHandlers(pMgr, &userdata, &handlers)) { \
116 SWL_CALL(handlers.fName, userdata, ifName); \
117 } \
118 }
119
120#define CALL_MGR_I_NA_EXT(pIntf, fName) \
121 if(pIntf != NULL) { \
122 char* ifName = (char*) wld_wpaCtrlInterface_getName(pIntf); \
123 CALL_MGR_NA_EXT(wld_wpaCtrlInterface_getMgr(pIntf), fName, ifName); \
124 }
125
126#define NOTIFY_EXT(PIFACE, FNAME, ...) \
127 if((PIFACE != NULL)) { \
128 /* for VAP/EP events */ \
129 CALL_INTF_EXT(PIFACE, FNAME, __VA_ARGS__); \
130 /* for RAD/Glob events */ \
131 CALL_MGR_I_EXT(PIFACE, FNAME, __VA_ARGS__); \
132 }
133
134#define NOTIFY_NA_EXT(PIFACE, FNAME, ...) \
135 if((PIFACE != NULL)) { \
136 /* for VAP/EP events */ \
137 CALL_INTF_NA_EXT(PIFACE, FNAME); \
138 /* for RAD/Glob events */ \
139 CALL_MGR_I_NA_EXT(PIFACE, FNAME); \
140 }
