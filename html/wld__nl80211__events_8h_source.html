195/*
196 * @brief create a listener for a set of nl80211 events, received from a selection of wiphy/iface.
197 * Received events are notified to the listener when:
198 * - listener has a dedicated valid handler function (i.e not null)
199 * - source wiphy/ifIndex match with the listener filter
200 * Wildcards (WLD_NL80211_ID_ANY) can be used to create a default wiphy or a global event listener.
201 * Exclusion (WLD_NL80211_ID_UNDEF) can be used to ignore events where expected source (wiphy or iface) is unknown.
202 * When multiple listeners are matching a received event+wiphy/ifIndex, they are invoked in this defined order:
203 * 1) first, the specific iface listeners
204 * 2) then, the default wiphy listeners (any iface)
205 * 3) last, the global listeners (any wiphy)
206 *
207 * @param state pointer to nl80211 socket manager, on which socket the event would be received
208 * @param wiphy Index of wiphy to listen for it (>=0):
209 * WLD_NL80211_ID_ANY for any wiphy
210 * WLD_NL80211_ID_UNDEF to exclude events where wiphy is included
211 * @param ifIndex Net dev index to listen for it (>0)
212 * WLD_NL80211_ID_ANY for any nl80211 interface
213 * WLD_NL80211_ID_UNDEF to exclude events where ifIndex is included
214 * @param pRef user private reference to provide when invoking handlers
215 * @param pData user private data to provide when invoking handlers
216 * @param handlers Pointer to structure of callback functions, handling the received events
217 * Callbacks of unmanaged events shall remain null.
218 *
219 * @return pointer to created listener
220 * It can be deleted by caller or implicitly when state is cleaned up.
221 */
222wld_nl80211_listener_t* wld_nl80211_addEvtListener(wld_nl80211_state_t* state,
223 uint32_t wiphy, uint32_t ifIndex,
224 void* pRef, void* pData,
225 const wld_nl80211_evtHandlers_cb* const handlers);
