147/* wld daemon context. */
148struct wld_process {
149 char* cmd; /* Command to run */
150 char* args; /* arguments as provided by user */
151 uint32_t argLen; /* length of args */
152 char** argList; /* argument list to be provided to start API */
153 uint32_t nrArgs; /* number of arguments */
154 bool enabled; /* Indicates if daemon is enabled(true) or disabled(false) */
155 wld_deamonState_e status; /* Indicates daemon's current state */
156 uint32_t fails; /* Amount of fails since last time daemon is set to enabled */
157 uint32_t totalFails; /* Total amount of fails since first time daemon is set to enabled */
158 uint32_t restarts; /* Amount of restarts since last time daemon is set to enabled */
159 uint32_t totalRestarts; /* Total amount of restarts since first time daemon is set to enabled */
160 time_t failDate; /* Indicates the last time daemon has been started */
161 bool forceKill; /* Force -9 option when terminating, as done by some ref software */
162
163 amxp_proc_ctrl_t* process; /* Process information struct */
164 amxp_timer_t* restart_timer; /* used when a daemon need to be restarted */
165 int stdInPeer; /* peer for stdIn reading */
166 int stdErrPeer; /* peer for stdErr reading */
167
168 wld_deamonExitInfo_t lastExitInfo; /* Last process termination details. */
169 wld_deamonEvtHandlers handlers; /* Optional handlers of daemon process events. */
170 void* userData; /* Optional userdata for process */
171};
