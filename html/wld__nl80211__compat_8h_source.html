80/*
81 * This header file replaces with "define" all required nl80211 "enum", used in the code,
82 * that are potentially undefined in some old kernel version.
83 * Those already defined, are keeping same official values.
84 * In fact:
85 * nl80211 cmd and attribute IDs are defined in enum , not macros
86 * so DEFINE same enum values, if they are lower than enum MAX,
87 * else consider them by default UNSPECIFIED
88 */
89#define VAL(v, m, d) (((v) > (m)) ? (d) : (v))
90#define NL80211_CMD(c) VAL(c, NL80211_CMD_MAX, NL80211_CMD_UNSPEC)
91#define NL80211_ATTR(a) VAL(a, NL80211_ATTR_MAX, NL80211_ATTR_UNSPEC)
92#define NL80211_BAND_ATTR(a) VAL(a, NL80211_BAND_ATTR_MAX, __NL80211_BAND_ATTR_INVALID)
93#define NL80211_BAND_IFTYPE_ATTR(a) VAL(a, NL80211_BAND_IFTYPE_ATTR_MAX, __NL80211_BAND_IFTYPE_ATTR_INVALID)
94#define NL80211_FREQUENCY_ATTR(a) VAL(a, NL80211_FREQUENCY_ATTR_MAX, __NL80211_FREQUENCY_ATTR_INVALID)
95#define NL80211_RATE_ATTR(a) VAL(a, NL80211_RATE_INFO_MAX, __NL80211_RATE_INFO_INVALID)
96#define NL80211_STA_INFO(a) VAL(a, NL80211_STA_INFO_MAX, __NL80211_STA_INFO_INVALID)
97#define NL80211_STA_FLAG(a) VAL(a, NL80211_STA_FLAG_MAX, __NL80211_STA_FLAG_INVALID)
98#define NL80211_SURVEY_INFO(a) VAL(a, NL80211_SURVEY_INFO_MAX, __NL80211_SURVEY_INFO_INVALID)
99
100#define IS_SPECIFIED(v, m, u) (((v) <= (m)) && ((v) > (u)))
101#define NL80211_CMD_IS_SPECIFIED(a) IS_SPECIFIED(c, NL80211_CMD_MAX, NL80211_CMD_UNSPEC)
102#define NL80211_ATTR_IS_SPECIFIED(a) IS_SPECIFIED(a, NL80211_ATTR_MAX, NL80211_ATTR_UNSPEC)
103#define NL80211_BAND_ATTR_IS_SPECIFIED(a) IS_SPECIFIED(a, NL80211_BAND_ATTR_MAX, __NL80211_BAND_ATTR_INVALID)
104#define NL80211_BAND_IFTYPE_ATTR_IS_SPECIFIED(a) IS_SPECIFIED(a, NL80211_BAND_IFTYPE_ATTR_MAX, __NL80211_BAND_IFTYPE_ATTR_INVALID)
