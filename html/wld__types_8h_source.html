134 FSM_SYNC_RAD, /* Sync the Radio states --> T_Radio */
135 FSM_SYNC_VAP, /* Sync VAP interfaces --> T_AccessPoint & T_SSID */
136 FSM_DEPENDENCY, /* Task checks dependency on the to-do list */
137 FSM_SYNC_GLOBAL, /* do global sync of dependencies. Calling global function that will wait if returned true*/
138 FSM_RUN, /* Run through the to-do list */
139 FSM_COMPEND, /* Check if a new commit is pending? */
140 FSM_WAIT_RAD, /* Wait for other radio's to finish config */
141 FSM_FINISH, /* To-do list is empty, all went fine */
142 FSM_ERROR, /* We've failed to execute the to-do list */
143 FSM_UNKNOWN, /* ? A state we don't like and don't know how to get rid of it.*/
144 FSM_FATAL, /* Really hard issue... we can only fix this by restarting the system or driver */
145 FSM_MAX /* Max value, not actually handled in FSM */
146} FSM_STATE;
